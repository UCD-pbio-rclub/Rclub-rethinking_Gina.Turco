2M1:

Grid Approximate posterior dist for each of the followiung sets of observations assumuing uniform pior

1. W,W,W

w = 3
# define grid
p_grid <- seq( from=0 , to=1 , length.out=1000 )
# define prior, All priors are equal
prior <- rep( 1 , 1000 )
# compute likelihood at each value in grid
likelihood <- dbinom( w , size=3 , prob=p_grid )
# compute product of likelihood and prior
unstd.posterior <- likelihood * prior
# standardize the posterior, so it sums to 1
posterior <- unstd.posterior / sum(unstd.posterior)

plot( p_grid , posterior , type="b" ,
    xlab="probability of water" , ylab="posterior probability" )

2M2:

1. W,W,W

w = 3
# define grid
p_grid <- seq( from=0 , to=1 , length.out=1000 )
# p is equal to zero when < 0.5 and 1 when greater than .5

# compute product of likelihood and prior
